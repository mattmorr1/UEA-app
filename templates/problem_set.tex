\begin{problem}[10 points]
\textbf{Linear Algebra Basics}

Let $A = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix}$ and $B = \begin{pmatrix} 5 & 6 \\ 7 & 8 \end{pmatrix}$.

\begin{enumerate}[label=(\alph*)]
    \item Compute $AB$.
    \item Find the eigenvalues of $A$.
    \item Is $A$ invertible? If so, find $A^{-1}$.
\end{enumerate}
\end{problem}

\begin{solution}
% Your solution here
\textbf{(a)} Computing $AB$:
\[
AB = \begin{pmatrix} 1 & 2 \\ 3 & 4 \end{pmatrix} \begin{pmatrix} 5 & 6 \\ 7 & 8 \end{pmatrix} = \begin{pmatrix} 19 & 22 \\ 43 & 50 \end{pmatrix}
\]

\textbf{(b)} Eigenvalues of $A$:
The characteristic polynomial is $\det(A - \lambda I) = \lambda^2 - 5\lambda - 2 = 0$.
\[
\lambda = \frac{5 \pm \sqrt{33}}{2}
\]

\textbf{(c)} $\det(A) = 4 - 6 = -2 \neq 0$, so $A$ is invertible:
\[
A^{-1} = \frac{1}{-2}\begin{pmatrix} 4 & -2 \\ -3 & 1 \end{pmatrix} = \begin{pmatrix} -2 & 1 \\ 3/2 & -1/2 \end{pmatrix}
\]
\end{solution}
